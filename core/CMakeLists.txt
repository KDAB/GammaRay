  LINK_PUBLIC gammaray_common
  LINK_PRIVATE gammaray_common_internal
)

  ${GAMMARAY_DEFAULT_LIBRARY_PROPERTIES}
  OUTPUT_NAME gammaray_core-${GAMMARAY_PROBE_ABI}
)

    LINK_PRIVATE ${QT_QTNETWORK_LIBRARIES}
  )
